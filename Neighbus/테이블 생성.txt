-- 데이터베이스 생성
CREATE DATABASE neighbus DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

use neighbus;

-- 유저 등급 테이블
create table grades(
    id int primary key,
    grade varchar(3) not null
);

-- 도, 특별시 테이블
create table provinces(
    id int primary key,
    province varchar(50)
);

-- 시, 군, 구 테이블
create table cities(
    id int primary key,
    province int not null,
    city varchar(50) not null
);

-- 유저 테이블
create table users(
    id int auto_increment primary key,
    username varchar(20) not null unique,
    pwd varchar(50) not null,
    city int not null,
    address varchar(200) not null,
    phone varchar(11) not null unique,
    email varchar(50) not null unique,
    image varchar(100) null,
    grade int default 1 not null,
    birth varchar(6) not null,
    sex varchar(1) not null,
    user_uuid varchar(36) not null,
    nickname varchar(20) null,
    created_at datetime default now(),
    FOREIGN KEY (city) REFERENCES cities(id),
    FOREIGN KEY (grade) REFERENCES grades(id)
);

-- 친구 테이블
create table friends(
	my_id int not null,
    friend_id int not null,
    state int default 1,
    FOREIGN KEY (my_id) REFERENCES users(id),
    FOREIGN KEY (friend_id) REFERENCES users(id),
    primary key(my_id, friend_id)
);

-- 자유게시판 테이블
create table freeboards(
    id int auto_increment primary key,
    title varchar(100) not null,
    content text not null,
    writer int not null,
    view_count int default 0,
    created_at datetime default now(),
    FOREIGN KEY (writer) REFERENCES users(id)
);

-- 자유게시판 이미지 테이블
create table freeboard_images(
    id int auto_increment primary key,
    freeboard int not null,
    img varchar(50) not null,
    FOREIGN KEY (freeboard) REFERENCES freeboards(id)
);

-- 자유게시판 댓글 테이블
create table freeboard_comments(
    id int auto_increment primary key,
    freeboard int not null,
    parent int null,
    writer int not null,
    content varchar(500),
    created_at datetime default now(),
    FOREIGN KEY (freeboard) REFERENCES freeboards(id),
    FOREIGN KEY (writer) REFERENCES users(id)
);

-- 동아리 테이블
create table clubs(
    id int auto_increment primary key,
    club_name varchar(100) not null,
    creator int not null,
    city int not null,
    created_at datetime default now(),
    FOREIGN KEY (creator) REFERENCES users(id),
    FOREIGN KEY (city) REFERENCES cities(id)
);

-- 동아리 멤버 테이블
create table club_members(
	club_id int not null,
    user_id int not null,
    created_at datetime default now(),
    FOREIGN KEY (club_id) REFERENCES clubs(id),
    FOREIGN KEY (user_id) REFERENCES users(id),
    primary key(club_id, user_id)
);

-- 모집 테이블
create table recruitments(
    id int auto_increment primary key,
    club int not null,
    title varchar(100) not null,
    content text not null,
    writer int not null,
    address varchar(200) not null,
    max_user int not null,
    meeting_date datetime not null,
    created_at datetime default now(),
    FOREIGN KEY (club) REFERENCES clubs(id),
    FOREIGN KEY (writer) REFERENCES users(id)
);

-- 갤러리 테이블
create table galleries(
    id int auto_increment primary key,
    title varchar(100) not null,
    content text not null,
    writer int not null,
    view_count int default 0,
    created_at datetime default now(),
    FOREIGN KEY (writer) REFERENCES users(id)
);

-- 갤러리 이미지 테이블
create table gallery_images(
    id int auto_increment primary key,
    gallery int not null,
    img varchar(50) not null,
    FOREIGN KEY (gallery) REFERENCES galleries(id)
);

-- 갤러리 댓글 테이블
create table gallery_comments(
    id int auto_increment primary key,
    gallery int not null,
    parent int null,
    writer int not null,
    content varchar(50) not null,
    created_at datetime default now(),
    FOREIGN KEY (gallery) REFERENCES galleries(id),
    FOREIGN KEY (writer) REFERENCES users(id)
);

-- 문의 테이블
create table inquiries(
    id int auto_increment primary key,
    title varchar(100) not null,
    content text not null,
    writer int not null,
    state int default 1,
    created_at datetime default now(),
    FOREIGN KEY (writer) REFERENCES users(id)
);

-- 문의 댓글 테이블
create table inquiry_comments(
    id int auto_increment primary key,
    inquiry int not null,
    content text not null,
    writer int not null,
    created_at datetime default now(),
    FOREIGN KEY (inquiry) REFERENCES inquiries(id),
    FOREIGN KEY (writer) REFERENCES users(id)
);

-- 공지사항 테이블
create table notices(
    id int auto_increment primary key,
    title varchar(100) not null,
    content text not null,
    writer int not null,
    created_at datetime default now(),
    FOREIGN KEY (writer) REFERENCES users(id)
);
